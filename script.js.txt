const diagnostic = [
    {
        question: "Votre ordinateur s'allume-t-il ?",
        oui: 1,
        non: "Vérifiez le câble d'alimentation, la batterie ou le bouton d'alimentation."
    },
    {
        question: "L'écran affiche-t-il quelque chose ?",
        oui: 2,
        non: "Vérifiez le câble vidéo, l'écran ou la carte graphique."
    },
    {
        question: "Windows démarre-t-il normalement ?",
        oui: "Votre ordinateur semble fonctionner normalement.",
        non: "Essayez le mode sans échec ou une réparation du démarrage."
    }
];

let etape = 0;

function afficherQuestion() {

    const zone = document.getElementById("diagnostic");

    if (typeof diagnostic[etape] === "undefined") return;

    zone.innerHTML = `
        <h2>${diagnostic[etape].question}</h2>

        <button onclick="repondre('oui')">Oui</button>

        <button onclick="repondre('non')">Non</button>
    `;
}

function repondre(reponse){

    let resultat = diagnostic[etape][reponse];

    if(typeof resultat === "number"){

        etape = resultat;

        afficherQuestion();

    }else{

        document.getElementById("diagnostic").innerHTML = `
            <h2>Résultat du diagnostic</h2>

            <p>${resultat}</p>

            <button onclick="recommencer()">
                Recommencer
            </button>
        `;

    }

}

function recommencer(){

    etape = 0;

    afficherQuestion();

}

window.onload = afficherQuestion;
